def mystery(x):
    if not isinstance(x, int) or isinstance(x, bool):
        return False

    if x <= 0 or x > 10**9:
        return False

    import math

    # Condition A: x must be a perfect square
    r = math.isqrt(x)
    if r * r != x:
        return False

    # Condition B: the square root of x must itself be a triangular number
    # T_n = n*(n+1)/2 = r  iff  1 + 8r is a perfect square
    disc = 1 + 8 * r
    s = math.isqrt(disc)
    if s * s != disc or (s - 1) % 2 != 0:
        return False

    # Condition C: factorize x and inspect its prime factors
    n, primes = x, set()
    d = 2
    while d * d <= n:
        if n % d == 0:
            primes.add(d)
            while n % d == 0:
                n //= d
        d += 1
    if n > 1:
        primes.add(n)

    # Must have exactly 5 distinct prime factors
    if len(primes) != 5:
        return False

    # The distinct prime factors must sum to exactly 34
    if sum(primes) != 34:
        return False

    return True
