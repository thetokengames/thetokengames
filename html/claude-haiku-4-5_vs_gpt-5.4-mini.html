def mystery(x):
    if not isinstance(x, int) or isinstance(x, bool):
        return False
    
    if x <= 1:
        return False
    
    # Check if x is a perfect square
    sqrt_x = int(x ** 0.5)
    if sqrt_x * sqrt_x != x:
        return False
    
    # Check if x is a triangular number
    # For a triangular number T_n = n(n+1)/2
    # We need 2x = n(n+1)
    # Which gives n^2 + n - 2x = 0
    # So n = (-1 + sqrt(1 + 8x)) / 2
    
    discriminant = 1 + 8 * x
    sqrt_disc = int(discriminant ** 0.5)
    if sqrt_disc * sqrt_disc != discriminant:
        return False
    
    # Check if (-1 + sqrt_disc) is even
    if (-1 + sqrt_disc) % 2 != 0:
        return False
    
    n = (-1 + sqrt_disc) // 2
    if n <= 0:
        return False
    
    return True
