    if not isinstance(x, int) or isinstance(x, bool):
        return False

    if x <= (1 << 455) or x >= (1 << 456):
        return False

    p = (1 << 256) - (1 << 32) - 977
    a = 1

    gx = 3
    gy = pow(30, (p + 1) // 4, p)
    G = (gx, gy)

    def inv(z):
        return pow(z % p, p - 2, p)

    def add(P, Q):
        if P is None:
            return Q
        if Q is None:
            return P

        x1, y1 = P
        x2, y2 = Q

        if x1 == x2 and (y1 + y2) % p == 0:
            return None

        if P == Q:
            m = ((3 * x1 * x1 + a) * inv(2 * y1)) % p
        else:
            m = ((y2 - y1) * inv(x2 - x1)) % p

        xr = (m * m - x1 - x2) % p
        yr = (m * (x1 - xr) - y1) % p
        return (xr, yr)

    def mul(k, P):
        R = None
        while k:
            if k & 1:
                R = add(R, P)
            P = add(P, P)
            k >>= 1
        return R

    return mul(x, G) == G
