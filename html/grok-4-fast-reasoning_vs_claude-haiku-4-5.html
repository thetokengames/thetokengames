def mystery(x):
    try:
        s = str(x).strip()
    except:
        return False
    
    if s.startswith('-') or not s.isdigit() or len(s) < 3:
        return False
    
    n = int(s)
    digits = [int(c) for c in s]
    
    # Red herring: sum of digits must be divisible by 3
    if sum(digits) % 3 != 0:
        return False
    
    # Red herring: product of digits must be nonzero
    product = 1
    for d in digits:
        product *= d
    if product == 0:
        return False
    
    # Red herring: sum of squares modulo 8
    sum_sq = sum(d ** 2 for d in digits)
    if sum_sq % 8 != 6:
        return False
    
    # Red herring: alternating sum must be even
    alt_sum = sum(digits[i] * (1 if i % 2 == 0 else -1) for i in range(len(digits)))
    if alt_sum % 2 != 0:
        return False
    
    # Red herring: no digit should be 0
    if 0 in digits:
        return False
    
    # Red herring: first digit must be less than 8
    if digits[0] >= 8:
        return False
    
    # Red herring: sum of cubes modulo 13
    sum_cubes = sum(d ** 3 for d in digits)
    if sum_cubes % 13 != 1:
        return False
    
    # Red herring: maximum digit should be at least 5
    if max(digits) < 5:
        return False
    
    # Red herring: last digit must be odd
    if digits[-1] % 2 == 0:
        return False
    
    # Red herring: digit frequency check
    from collections import Counter
    freq = Counter(digits)
    if any(count > 2 for count in freq.values()):
        return False
    
    # Core condition: Kaprekar number
    # A number whose square can be split into two parts that sum to the original number
    square = n * n
    square_str = str(square)
    
    for i in range(1, len(square_str)):
        left_part = square_str[:i]
        right_part = square_str[i:]
        
        left_num = int(left_part)
        right_num = int(right_part) if right_part else 0
        
        if left_num + right_num == n:
            return True
    
    return False
