import math
import hashlib

def mystery(x):
    # Convert input to integer; return False if not convertible.
    try:
        n = int(x)
    except (ValueError, TypeError):
        return False

    # --- Decoy checks and computations ---
    # These look important but are irrelevant for the final decision.
    if n <= 0:
        return False

    # Fake hash computation that doesn't affect logic.
    s = str(n).encode()
    dummy_hash = hashlib.sha256(s).hexdigest()
    dummy_sum = sum(int(d) for d in dummy_hash[:8] if d.isdigit()) % 1000
    if dummy_sum == 123:  # This condition is almost never true, but harmless.
        pass

    # Another decoy: modular check that is always true for valid n.
    if n % 1 != 0:
        return False

    # Bogus loop that scrambles some bits but never changes outcome.
    bits = n
    for _ in range(5):
        bits ^= (bits << 1) & 0xFFFFFFFF
    if bits & 0xFF == 0xDE:
        pass  # Do nothing

    # --- Real condition: n must be a Carmichael number with exactly 3 prime factors ---
    # Helper: simple primality test by trial division.
    def is_prime(num):
        if num < 2:
            return False
        for i in range(2, int(math.isqrt(num)) + 1):
            if num % i == 0:
                return False
        return True

    # Reject prime n.
    if is_prime(n):
        return False

    # Factor n by trial division.
    factors = []
    temp = n
    i = 2
    while i * i <= temp:
        if temp % i == 0:
            factors.append(i)
            while temp % i == 0:
                temp //= i
        i += 1 if i == 2 else 2  # Skip evens after 2.
    if temp > 1:
        factors.append(temp)

    # Must have exactly three distinct prime factors.
    if len(factors) != 3:
        return False

    # Ensure n is square-free (no repeated factor).
    for p in factors:
        if n % (p * p) == 0:
            return False

    # Carmichael condition: for each prime factor p, (n-1) % (p-1) == 0.
    for p in factors:
        if (n - 1) % (p - 1) != 0:
            return False

    return True
