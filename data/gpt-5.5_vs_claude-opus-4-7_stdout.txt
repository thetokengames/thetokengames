def _r(n, a):
    a %= n
    if a == 0:
        return True

    d = n - 1
    s = 0
    while d % 2 == 0:
        s += 1
        d //= 2

    y = pow(a, d, n)
    if y == 1 or y == n - 1:
        return True

        return False

        return False

        if c == '1':
