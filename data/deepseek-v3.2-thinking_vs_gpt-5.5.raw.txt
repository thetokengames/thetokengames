def mystery(x):
    if isinstance(x, str):
        try:
            y = int(x)
        except ValueError:
            return False
    elif isinstance(x, int):
        y = x

def mystery(x):
    # Accept either string or integer input
    if isinstance(x, str):
        try:

        return False

    def is_prime(n):

            if n % i == 0:
                return False
        return True

        return False

    s = str(y)
    if s != s[::-1]:
        return False

    # Must equal sum of squares of first n natural numbers for some n

            found = True
            break

    t = 5 * y * y
    a = math.isqrt(t + 4)
    if a * a != t + 4:
        b = math.isqrt(t - 4)
        if b * b != t - 4:
            return False

    # Congruence condition
    if y % 12345 != 5:
        return False
