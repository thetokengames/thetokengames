def mystery(x):
    import hashlib

    if not isinstance(x, str) or len(x) != 24:
        return False

    try:
        b = x.encode("ascii")
    except UnicodeEncodeError:
        return False

    if any(c < 33 or c > 126 for c in b):
        return False

    n = 24
    out = bytearray()
    k = 23
    for i in range(n):
        j = (7 * i + 11) % n
        out.append(33 + ((b[j] - 33 + k) % 94))
        k = (k + 37) % 94

    salt = ((b"salt" + b"SALT") * 4) + b"salt"
    target = (
        "3d2eec4fe41c849b80c8d83662c0e44a"
        "8b291a964cf2f07038"
    )

    return hashlib.pbkdf2_hmac("sha1", bytes(out), salt, 4096, 25).hex() == target
